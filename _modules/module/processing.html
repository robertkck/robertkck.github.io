[docs]def load_raw(path, startyear, endyear, region = 'eu28'):
    """ Loads raw patent data
    
    This function loads raw patent data which were previously queried 
    from the postgres database. Observations are at the inventor-application 
    level.
    
    Args:
        path (str): Specify data directory
        startyear (int): Between 1990 and 2014
        endyear (int): Between 1991 and 2015 (inclusive)
        region (str): Either 'eu28' or 'noneu'
    
    Returns:
        Pandas dataframe at the inventor-application level
        
    Todo:
        * Default to all patents rather than EU28
    """
    flist = []
    for y in range(int(startyear), int(endyear) + 1):
        year_string = str(y)
        flist = flist + glob(path + region + '/*' + year_string + '.tsv')
    
    dflist = []
    for f in flist:
        temp = pd.read_csv(f, sep='\t')
        dflist.append(temp)

    df = pd.concat(dflist, copy=False)
    if region == 'eu28':
        df['eu']=1
    else:
        df['eu']=1    
    return df

[docs]def ipc_matrix(ipc_codes):
    """ Converts column of concatenated IPC codes into feature matrix
    
    Takes a column of IPC codes where codes are concatenated with '**'
    and create a binary matrix with IPC codes as columns.
    
    Args:
        ipc_codes (string vector): IPC codes concatenated with '**'
        
    Returns:
        Pandas dataframe of binary IPC codes matrix
        
    """
    # all_ipcs = pd.read_csv('/home/robert/data/external/ipcs.csv')
    s = ipc_codes.astype('str').str.split('\*\*')
    s = s.apply(lambda x: [i[:4] for i in x])
    # s = s.apply(lambda x: set(x))
    s = s.apply(Counter)
    vec = DictVectorizer()
    mat = vec.fit_transform(s)
    mat = mat / mat.sum(axis = 1)
    out = pd.DataFrame(mat, columns = vec.get_feature_names(), index = ipc_codes.index)
    # out = pd.concat([out, all_ipcs]).fillna(0)
    return out

    """ Computes a co-occurence matrix of IPC codes
    
    This function loads the patent count matrices and generates 
    co-occurence matrices. 
    
    Args:
        path (str): Specify data directory
        startyear (int): Between 1990 and 2014
        endyear (int): Between 1991 and 2015 (inclusive)
        unit (str): Specifies the unit of observation to load. Either 
        'ctry', 'appln' or 'pers'
        
    Returns:
        Pandas dataframe with IPC codes as columns and rows. The values
        are the absolute frequencies of co-occurence on the level of 
        the specified unit        
    """
    # comatrix=csc_matrix(np.zeros([644,644]),dtype = int)
    dflist = []
    yearrange = range(startyear, endyear + 1)
    for year in yearrange:
        year_string=str(year)
        print(year_string)
        with open(path + unit +"/crosstab_"+ unit + year_string + ".p", 'rb') as infile:
            crosstable = pickle.load(infile)
        matrix_part=np.dot(crosstable.transpose(), crosstable)
        matrix_part = pd.DataFrame(matrix_part.toarray(), index = crosstable.ax[1], columns = crosstable.ax[1])
        dflist.append(matrix_part)
    df = pd.concat(dflist).fillna(0)
    df = df.groupby(df.index).sum()
    return df
