[docs]def load_exports(t0, t1, path = '/home/robert/syntax/comtrade/'):
    list_ = []
    yearrange = range(t0, t1 + 1)    
    for year in yearrange:
        year_string = str(year)
        loadloc = path + '/data/yearly/AG6/ExtoWorld_AG6_%s.csv' % (year_string)
        df=pd.read_csv(
            loadloc, 
            sep = "," , 
            usecols = ['Reporter ISO', 'Commodity Code', 'Trade Value (US$)'],
            dtype = {'Reporter ISO':str, 'Commodity Code':str, 'Trade Value (US$)': np.int64})
        list_.append(df)
    trade = pd.concat(list_)
    trade.rename(columns = {'Reporter ISO':'ctry', 'Commodity Code':'hs', 'Trade Value (US$)':'value'}, inplace = True)
    trade = trade.groupby(['ctry', 'hs']).sum().fillna(0)
    return trade
