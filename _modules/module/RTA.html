[docs]def compute(patents, *standard):
    RTA = patents.divide(patents.sum(axis = 1), axis = 0)
    RTA = RTA.divide(patents.sum().divide(patents.sum().sum()))
    if ('s' in standard):
        print("standardised RTA")
        RTA = (((RTA - 1)/(RTA + 1)) + 1) / 2
    return RTA
